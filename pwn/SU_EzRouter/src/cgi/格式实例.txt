struct __attribute__((packed)) vpn_config_req {
    char action[16];    // 动作 (e.g., "apply")
    char name[32];      // 连接名称
    char proto[16];     // 协议类型
    char server[64];    // 服务器地址
    char user[64];      // 用户名
    char pass[64];      // 密码
    char cert[64];      // 证书信息
    char custom[512];   // 自定义高级选项
} req;
----------------------------------------------------------------
struct { 
    char mac[0x40];     // MAC 地址, 64字节
    char note[0x300];   // 备注说明, 768字节
} req;
----------------------------------------------------------------
struct {
    char ssid[64];      // Wi-Fi 名称
    char password[64];  // Wi-Fi 密码
} req;
